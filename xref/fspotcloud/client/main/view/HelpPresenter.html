42  	private String getHelpGroup(ActionMap group) {

44  		String helptext = "<table style='text-align: left'>";
45  		helptext += "<tr><th colspan=4 style='text-align: center'>";
46  		helptext += title;
47  		helptext += "</th></tr></th>";
48  		for (UserAction shortcut : group.allActions()) {
49  			helptext += "<tr><td>" + generator.getHelpText(shortcut)
50  					+ "</td></tr>";
51  		}
52  		helptext += "</table>";

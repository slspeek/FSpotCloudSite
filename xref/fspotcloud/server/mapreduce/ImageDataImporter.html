49  	private boolean isRelevant(Photo photo) {

51  		for (String wantedTag : tagsForImport) {

53  				return true;
54  			}
55  		}
56  		return false;
57  	}

59  	private void scheduleImage() {
60  		if (!photo.isImageLoaded() && isRelevant(photo)) {
61  			List<String> args = new ArrayList<String>();

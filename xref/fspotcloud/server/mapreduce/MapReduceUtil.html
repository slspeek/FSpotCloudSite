43  	public static String startJob(String jobName, String kind) {
44  		TaskOptions task = buildStartJob(jobName, kind);
45  		String url = task.getUrl();
46  		URLFetchService fetcher = URLFetchServiceFactory.getURLFetchService();
47  		HTTPResponse response = null;
48  		HTTPRequest request = null;
49  		String host = "localhost:8080";
50  		String port = "80";
51  		try {
52  			host = InetAddress.getLocalHost().getHostName();
53  			port = host.equals("localhost") ? "8080" : "80";
54  		} catch (UnknownHostException e1) {
55  			log.log(Level.SEVERE, "could not retrieve hostname", e1);
56  		}
57  		try {
58  			request = new HTTPRequest(new URL("http://" + host + ":" + port
59  					+ url));
60  			request.addHeader(new HTTPHeader("X-Requested-With",
61  					"XMLHttpRequest"));
62  			response = fetcher.fetch(request);
63  		} catch (Exception e) {
64  			log.log(Level.SEVERE, "startJob failed", e);
65  		}
66  		String result = response != null ? new String(response.getContent())
67  				: "NO CONTENT";
68  		return result;
69  	}

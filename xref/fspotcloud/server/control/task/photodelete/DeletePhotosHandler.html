41  			throws DispatchException {
42  		Tag tag = tagManager.getById(action.getTagId());
43  		Iterator<PhotoInfo> it = action.getToBoDeleted().iterator();
44  		for (int i = 0; i < MAX_DELETE_TICKS && it.hasNext(); i++) {
45  			String key = it.next().getId();
46  			checkForDeletion(tag, tag.getId(), key, it);
47  		}
48  		tagManager.save(tag);
49  		if (!action.getToBoDeleted().isEmpty()) {
50  			dispatchAsync.execute(action);
51  		}
52  		return new VoidResult();
53  	}

55  	private void checkForDeletion(Tag tag, String deleteTagId, String key,
56  			Iterator<PhotoInfo> it) {
57  		Photo photo = photos.getById(key);
58  		if (photo != null) {
59  			boolean moreImports = false;
60  			for (String tagId : photo.getTagList()) {
61  				Tag tagRelated = tagManager.getById(tagId);
62  				if (tagRelated != null) {
63  					if (!deleteTagId.equals(tagId)) {
64  						if (tagRelated.isImportIssued()) {
65  							moreImports = true;
66  							break;
67  						}
68  					}
69  				}
70  			}
71  			if (!moreImports) {
72  				photos.delete(key);
73  				tag.getCachedPhotoList().remove(find(tag.getCachedPhotoList(), key));
74  			}
75  		}
76  		it.remove();
77  		
78  	}

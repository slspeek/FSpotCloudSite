140 	public URL getImageURL(String photoId) throws SQLException,
141 			MalformedURLException {
142 		String url = null;
143 		Connection conn = getConnection();
144 		Statement stmt = conn.createStatement();
145 		String query = "SELECT default_version_id, base_uri, filename "
146 				+ "FROM photos WHERE id = " + photoId;
147 		ResultSet rs = stmt.executeQuery(query);
148 		if (rs.next()) {
149 			int version = rs.getInt(1);
150 			if (version == 1) {
151 				url = rs.getString(2) + "/" + rs.getString(3);
152 			} else {
153 				stmt = conn.createStatement();
154 				query = "SELECT base_uri, filename "
155 						+ "FROM photo_versions WHERE photo_id =" + photoId
156 						+ " AND version_id=" + version;
157 				rs = stmt.executeQuery(query);
158 				if (rs.next()) {
159 					url = rs.getString(1) + "/" + rs.getString(2);
160 				}
161 			}
162 		}
163 		rs.close();
164 		conn.close();
165 		if (photoDirectoryOverride != null) {
166 			url = url.replaceFirst(photoDirectoryOriginalPath,
167 					photoDirectoryOverride);
168 		}
169 		// log.info("URL-String: " + url);
170 		return new URL(url);
171 	}
